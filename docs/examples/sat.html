    <script>
      var app = initApp('#container');
      var basis = generateBasis(app);

      drawGrid(app, basis);

      function initApp(id) {
        var container = document.querySelector(id);
        var width = container.clientWidth;
        var height = container.clientHeight;
        var app = new PIXI.Application(width, height, {backgroundColor: 0x282a36});

        container.appendChild(app.view);
        return app;
      }

      function generateBasis(app) {
        var ref = app.screen;
        var width = ref.width;
        var height = ref.height;
        return new PIXI.Matrix(1, 0, 0, -1, width * 0.5, height * 0.5);
      }

      function drawGrid(app, basis) {
        var graphics = new PIXI.Graphics();
        var ref = app.screen;
        var width = ref.width;
        var height = ref.height;

        graphics.lineStyle(1, 0xf8f8f2, 1);
        graphics.moveTo(0, height * 0.5);
        graphics.lineTo(width, height * 0.5);
        graphics.moveTo(width * 0.5, 0);
        graphics.lineTo(width * 0.5, height);

        var point = new PIXI.Point(100, 100);

        var p = basis.apply(point);
        graphics.lineStyle(0); // draw a circle, set the lineStyle to zero so the circle doesn't have an outline
        graphics.beginFill(0xDE3249, 1);
        graphics.drawCircle(p.x, p.y, 50);
        graphics.endFill();

        app.stage.addChild(graphics);
      }
    </script>
