    <script>
      import {randomInt} from 'pg/utils/math';

      var GRAVITY = 1;
      var FRICTION = 0.79;

      var canvas = document.querySelector('canvas');
      var ctx = canvas.getContext('2d');
      var mouse = {
        x: undefined,
        y: undefined
      };
      var colorSet = [
        '#8be9fd',
        '#50fa7b',
        '#ffb86c',
        '#ff79c6',
        '#bd93f9',
        '#ff5555',
        '#f1fa8c' ];

      canvas.addEventListener('mousemove', function (event) {
        mouse.x = event.x;
        mouse.y = event.y;
      });

      canvas.addEventListener('click', function () {
        init();
      });

      window.addEventListener('resize', function () {
        resize();
      });

      var Ball = function Ball(x, y, dx, dy, radius, color) {
        this.x = x;
        this.y = y;
        this.dx = dx;
        this.dy = dy;
        this.radius = radius;
        this.color = color;
      };

      Ball.prototype.draw = function draw () {
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2, false);
        ctx.fillStyle = this.color;
        ctx.fill();
      };

      Ball.prototype.update = function update () {
        if (this.x + this.radius + this.dx >= canvas.width || this.x - this.radius + this.dx < 0) {
          this.dx *= -1 * FRICTION;
        }

        if (this.y + this.radius + this.dy >= canvas.height) {
          this.dy *= -1 * FRICTION;
          this.dx *= FRICTION;
        } else {
          this.dy += GRAVITY;
        }

        this.x += this.dx;
        this.y += this.dy;
        this.draw();
      };

      var balls = [];

      function init() {

        balls = [];

        for (var i = 0; i < 500; i++) {
          var radius = randomInt(8, 20);
          var x = randomInt(radius, window.innerWidth - radius);
          var dx = randomInt(-20, 20);
          var y = randomInt(radius, window.innerHeight - radius);
          var dy = randomInt(-2, 2);
          var color = colorSet[randomInt(0, colorSet.length)];

          balls.push(new Ball(x, y, dx, dy, radius, color));
        }
      }

      function resize() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
      }

      function animate() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        ctx.fillStyle = '#282a36';
        ctx.fillRect(0, 0, canvas.width, canvas.height);
        balls.forEach(function (circle) { return circle.update(); });
        requestAnimationFrame(animate);
      }

      resize();
      init();
      animate();
    </script>
  </body>
