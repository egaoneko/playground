  <head>
    <title>panzoom matrix</title>
    <link rel="stylesheet" href="http://www.3daysofprogramming.com/playground/pg.css" type="text/css">
    <!-- The line below is only needed for old environments like Internet Explorer and Android 4.x -->
    <script src="https://cdn.polyfill.io/v2/polyfill.min.js?features=requestAnimationFrame,Element.prototype.classList,URL,Map,Set,Promise"></script>
    <script src="https://unpkg.com/panzoom@7.1.2/dist/panzoom.min.js"></script>
    <style>
      #container {
        overflow: hidden;
      }

      #scene {
        width: 100%;
        height: 100%;
      }

      #circle {
        position: absolute;
        left: 10px;
        top: 10px;
      }
    </style>
  </head>
  <body>
    <div id="container" class="main">
      <div id="scene">
        <img id="content" src="https://www.filepicker.io/api/file/WsU90V7LTG8ez9OduzUv/convert?cache=true&crop=0%2C22%2C1278%2C639">
      </div>
    </div>

    <button id="get-center">Get Center</button>
    <button id="move-to-center">Move to Center</button>
    <script>
      var scene = document.querySelector('#scene');
      var content = document.querySelector('#content');
      var pz = panzoom(scene, {smoothScroll: false});
      var svg = document.createElementNS("http://www.w3.org/2000/svg", "svg");
      var basis = svg.createSVGMatrix();
      var sRect = scene.getBoundingClientRect();
      var cRect = content.getBoundingClientRect();

      function getBasis() {
        var ref = pz.getTransform();
        var x = ref.x;
        var y = ref.y;
        var scale = ref.scale;
        basis.a = scale;
        basis.b = 0;
        basis.c = 0;
        basis.d = scale;
        basis.e = x;
        basis.f = y;
        return basis;
      }

      function c2w(x, y) {
        var point = svg.createSVGPoint();
        point.x = x;
        point.y = y;

        var basis = getBasis();
        return point.matrixTransform(basis.inverse());
      }

      function w2c(x, y) {
        var point = svg.createSVGPoint();
        point.x = x;
        point.y = y;

        var basis = getBasis();
        return point.matrixTransform(basis);
      }

      document.querySelector('#get-center')
        .addEventListener('click', function () {
          var x = sRect.width * 0.5;
          var y = sRect.height * 0.5;
          console.log(c2w(x, y));
        });

      document.querySelector('#move-to-center')
        .addEventListener('click', function () {
          var x = cRect.width * 0.5;
          var y = cRect.height * 0.5;
          var point = w2c(x, y);
          pz.moveBy(sRect.width * 0.5 - point.x, sRect.height * 0.5 - point.y);
        });

      window['pz'] = pz;
    </script>
  </body>
