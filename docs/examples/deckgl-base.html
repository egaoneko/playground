    <script>
      var accessToken =
        "Your Mapbox access token from http://mapbox.com/ here";
      var container = document.querySelector("#map");

      var ambientLight = new deck.AmbientLight({
        color: [255, 255, 255],
        intensity: 1.0
      });

      var pointLight1 = new deck.PointLight({
        color: [255, 255, 255],
        intensity: 0.8,
        position: [-0.144528, 49.739968, 80000]
      });

      var pointLight2 = new deck.PointLight({
        color: [255, 255, 255],
        intensity: 0.8,
        position: [-3.807751, 54.104682, 8000]
      });

      var lightingEffect = new deck.LightingEffect({
        ambientLight: ambientLight,
        pointLight1: pointLight1,
        pointLight2: pointLight2
      });

      var material = {
        ambient: 0.64,
        diffuse: 0.6,
        shininess: 32,
        specularColor: [51, 51, 51]
      };

      var INITIAL_VIEW_STATE = {
        longitude: -1.4157267858730052,
        latitude: 52.232395363869415,
        zoom: 6.6,
        minZoom: 5,
        maxZoom: 15,
        pitch: 40.5,
        bearing: -27.396674584323023
      };

      var colorRange = [
        [1, 152, 189],
        [73, 227, 206],
        [216, 254, 181],
        [254, 237, 177],
        [254, 173, 84],
        [209, 55, 78]
      ];

      var deckgl = new deck.DeckGL({
        container: container,
        controller: true,
        reuseMaps: true,
        mapboxApiAccessToken: accessToken,
        preventStyleDiffing: true,
        mapStyle: "mapbox://styles/mapbox/dark-v9",
        initialViewState: INITIAL_VIEW_STATE
      });

      function renderLayer(props) {
        var data = props.data;
        var radius = props.radius; if ( radius === void 0 ) radius = 1000;
        var upperPercentile = props.upperPercentile; if ( upperPercentile === void 0 ) upperPercentile = 100;
        var coverage = props.coverage; if ( coverage === void 0 ) coverage = 1;

        var hexagonLayer = new deck.HexagonLayer({
          id: "heatmap",
          colorRange: colorRange,
          coverage: coverage,
          data: data,
          elevationRange: [0, 3000],
          elevationScale: data && data.length ? 50 : 0,
          extruded: true,
          getPosition: function (d) { return d; },
          radius: radius,
          upperPercentile: upperPercentile,
          material: material,

          transitions: {
            elevationScale: 3000
          }
        });

        deckgl.setProps({
          layers: [hexagonLayer],
          effects: [lightingEffect]
        });
      }

      d3.csv(
        'https://raw.githubusercontent.com/uber-common/deck.gl-data/master/examples/3d-heatmap/heatmap-data.csv'
      ).then(function (data) {
        data = data.map(function (d) { return [Number(d.lng), Number(d.lat)]; });
        renderLayer({
          data: data
        });
      });    </script>
