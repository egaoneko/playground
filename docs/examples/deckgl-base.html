    <div id="container" class="main">
      <div id="map">

    <script>
      var accessToken = 'Your Mapbox access token from http://mapbox.com/ here';
      var container = document.querySelector('#map');
      var deckgl = new deck.DeckGL({
        container: container,
        mapboxApiAccessToken: accessToken,
        mapStyle: 'mapbox://styles/mapbox/dark-v9',
        longitude: -1.4157,
        latitude: 52.2324,
        zoom: 6,
        minZoom: 5,
        maxZoom: 15,
        pitch: 40.5
      });

      var data = d3.csv('https://raw.githubusercontent.com/uber-common/deck.gl-data/master/examples/3d-heatmap/heatmap-data.csv');

      var COLOR_RANGE = [
        [1, 152, 189],
        [73, 227, 206],
        [216, 254, 181],
        [254, 237, 177],
        [254, 173, 84],
        [209, 55, 78]
      ];

      renderLayer();

      function renderLayer () {
        var hexagonLayer = new deck.HexagonLayer({
          id: 'heatmap',
          colorRange: COLOR_RANGE,
          data: data,
          elevationRange: [0, 1000],
          elevationScale: 250,
          extruded: true,
          getPosition: function (d) { return [Number(d.lng), Number(d.lat)]; },
          opacity: 1,
        });

        deckgl.setProps({
          layers: [hexagonLayer]
        });
      }    </script>
