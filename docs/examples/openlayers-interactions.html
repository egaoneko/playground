<!DOCTYPE html>
<html>
  <head>
    <title>Openlayers interactions</title>
    <link rel="stylesheet" href="http://www.3daysofprogramming.com/playground/pg.css" type="text/css">
    <!-- The line below is only needed for old environments like Internet Explorer and Android 4.x -->
    <script src="https://cdn.polyfill.io/v2/polyfill.min.js?features=requestAnimationFrame,Element.prototype.classList,URL"></script>
    <link rel="stylesheet" href="https://cdn.rawgit.com/openlayers/openlayers.github.io/master/en/v5.2.0/css/ol.css">
    <script src="https://cdn.rawgit.com/openlayers/openlayers.github.io/master/en/v5.2.0/build/ol.js"></script>
    <style>
      /**
        * Create a position for the map
        * on the page */
      #map {
        width: 100%;
        height: 100%;
      }
    </style>
  </head>
  <body>
    <div id="container" class="main">
      <div id="map" tabindex="0"></div>
    </div>
    <script>
      var raster = new ol.layer.Tile({
        source: new ol.source.OSM()
      });

      var container = document.querySelector('#map');

      var CustomFocus = (function (superclass) {
        function CustomFocus() {
          superclass.call(this, {
            handleDownEvent: handleDownEvent,
          });
        }

        if ( superclass ) CustomFocus.__proto__ = superclass;
        CustomFocus.prototype = Object.create( superclass && superclass.prototype );
        CustomFocus.prototype.constructor = CustomFocus;

        return CustomFocus;
      }(ol.interaction.Pointer));

      function handleDownEvent() {
        container.focus();
      }

      var map = new ol.Map({
        interactions: ol.interaction.defaults({
          doubleClickZoom: false,
          dragAndDrop: false,
          dragPan: false,
          keyboardPan: false,
          keyboardZoom: false,
          mouseWheelZoom: false,
          pointer: false,
          select: false
        }).extend([
          new CustomFocus({
            condition: function () { return true; }
          }),
          new ol.interaction.DragPan({
            kinetic: new ol.Kinetic(-0.02, 3 / 150, 100),
            condition: function () { return true; }
          }),
          new ol.interaction.KeyboardPan({
            condition: function () { return true; }
          })
        ]),
        layers: [raster],
        target: 'map',
        view: new ol.View({
          center: [-11000000, 4600000],
          zoom: 4
        }),
        keyboardEventTarget: container
      });
    </script>
  </body>
</html>
