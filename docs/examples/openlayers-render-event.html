    <title>Openlayers render event</title>
    <link rel="stylesheet" href="http://www.3daysofprogramming.com/playground/pg.css" type="text/css">
    <!-- The line below is only needed for old environments like Internet Explorer and Android 4.x -->
    <script src="https://cdn.polyfill.io/v2/polyfill.min.js?features=requestAnimationFrame,Element.prototype.classList,URL,Map,Set,Promise"></script>
    <link rel="stylesheet" href="https://cdn.rawgit.com/openlayers/openlayers.github.io/master/en/v5.2.0/css/ol.css">
    <script src="https://cdn.rawgit.com/openlayers/openlayers.github.io/master/en/v5.2.0/build/ol.js"></script>
    <style>
      /**
        * Create a position for the map
        * on the page */
      #map {
        width: 100%;
        height: 100%;
      }
    </style>
  </head>
  <body>
    <div id="container" class="main">
      <div id="map"></div>
    </div>
    <button id="add-feature">Add feature</button>
    <button id="add-layer">Add layer</button>
    <button id="refresh-source">Refresh source</button>
    <button id="set-zoom">Set zoom</button>
    <script>
      var source1 = new ol.source.OSM();
      var rasterTile1 = new ol.layer.Tile({
        source: source1,
      });

      var source2 = new ol.source.OSM();
      var rasterTile2 = new ol.layer.Tile({
        source: source2,
      });

      var styleFunction = function (color) {
        return new ol.style.Style({
          image: new ol.style.Circle({
            radius: 10,
            fill: null,
            stroke: new ol.style.Stroke({
              color: color
            })
          })
        });
      };

      var source3 = new ol.source.Vector({
        features: (new ol.format.GeoJSON()).readFeatures({
          'type': 'FeatureCollection',
          'crs': {
            'type': 'name',
            'properties': {
              'name': 'EPSG:3857'
            }
          },
          'features': [{
            'type': 'Feature',
            'geometry': {
              'type': 'Point',
              'coordinates': getRandomCoordinates()
            }
          }]
        })
      });

      var vector1 = new ol.layer.Vector({
        source: source3,
        style: function () {
          console.log("%cvector1", "background-color: red; color: white");
          return styleFunction('magenta');
        }
      });

      var source4 = new ol.source.Vector({
        features: (new ol.format.GeoJSON()).readFeatures({
          'type': 'FeatureCollection',
          'crs': {
            'type': 'name',
            'properties': {
              'name': 'EPSG:3857'
            }
          },
          'features': [{
            'type': 'Feature',
            'geometry': {
              'type': 'Point',
              'coordinates': getRandomCoordinates()
            }
          }]
        })
      });

      var vector2 = new ol.layer.Vector({
        source: source4,
        style: function () {
          console.log("%cvector2", "background-color: red; color: white");
          return styleFunction('magenta');
        }
      });

      var vectorTile = new ol.layer.VectorTile({
        source: new ol.source.VectorTile({
          format: new ol.format.MVT(),
          url: 'https://basemaps.arcgis.com/v1/arcgis/rest/services/World_Basemap/VectorTileServer/tile/{z}/{y}/{x}.pbf'
        }),
        style: function () {
          console.log("%cvectorTile", "background-color: red; color: white");
          return styleFunction('red');
        }
      });

      var map = new ol.Map({
        layers: [rasterTile1, rasterTile2, vectorTile, vector1],
        target: 'map',
        view: new ol.View({
          center: [0, 0],
          zoom: 2
        })
      });

      document
        .querySelector('#add-feature')
        .addEventListener('click', function () {
          var format = new ol.format.GeoJSON({
            featureProjection: 'EPSG:3857'
          });
          var geojson = {
            'type': 'FeatureCollection',
            'features': [{
              'type': 'Feature',
              'geometry': {
                'type': 'Point',
                'coordinates': [Math.random() * 360 - 180, Math.random() * 180 - 90]
              }
            }]
          };
          vector1.getSource().addFeatures(format.readFeatures(geojson));
        });

      document
        .querySelector('#add-feature')
        .addEventListener('click', function () {
          var format = new ol.format.GeoJSON({
            featureProjection: 'EPSG:3857'
          });
          var geojson = {
            'type': 'FeatureCollection',
            'features': [{
              'type': 'Feature',
              'geometry': {
                'type': 'Point',
                'coordinates': getRandomCoordinates()
              }
            }]
          };
          vector1.getSource().addFeatures(format.readFeatures(geojson));
        });

      document
        .querySelector('#add-layer')
        .addEventListener('click', function () {
          map.removeLayer(vector2);
          map.addLayer(vector2);
        });

      document
        .querySelector('#refresh-source')
        .addEventListener('click', function () {
          vector1.getSource().refresh();
        });

      document
        .querySelector('#set-zoom')
        .addEventListener('click', function () {
          map.getView().setZoom(map.getView().getZoom());
        });

      // render event
      [
        ['precompose', '#001f3f'],
        ['postcompose', '#0074D9'],
        ['render', '#7FDBFF'],
        ['postrender', '#39CCCC'],
        ['rendercomplete', '#3D9970'],
        ['tileloadstart', '#2ECC40'],
        ['tileloadend', '#01FF70'],
        ['tileloaderror', '#FFDC00'] ].forEach(function (ref) {
        var type = ref[0];
        var typeColor = ref[1];

        [
          [map, 'map', '#0074D9'],
          [rasterTile1, 'rasterTile1', '#7FDBFF'],
          [rasterTile2, 'rasterTile2', '#01FF70'],
          [vectorTile, 'vectorTile', '#39CCCC'],
          [vector1, 'vector1', '#3D9970'],
          [vector2, 'vector2', '#2ECC40'],
          [source1, 'source1', '#FFDC00'],
          [source2, 'source2', '#FF851B'],
          [source3, 'source3', '#FF4136'],
          [source4, 'source4', '#85144b'] ].forEach(function (ref) {
          var context = ref[0];
          var contextName = ref[1];
          var contextColor = ref[2];

          context.on(type, function () {
            log([contextName, contextColor], [type, typeColor]);
          });
        });
      });

      function log(ref, ref$1) {
        var contextName = ref[0];
        var contextColor = ref[1];
        var type = ref$1[0];
        var typeColor = ref$1[1];

        console.log(("%c[" + contextName + "] %c" + type), ("color: " + contextColor), ("color: " + typeColor));
      }

      function getRandomCoordinates() {
        return [Math.random() * 360 - 180, Math.random() * 180 - 90];
      }

      window.DEBUG = {
        map: map,
        rasterTile1: rasterTile1,
        rasterTile2: rasterTile2,
        vectorTile: vectorTile,
        vector1: vector1,
        vector2: vector2
      };
    </script>
  </body>
